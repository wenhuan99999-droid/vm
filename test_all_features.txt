// ============================================================
// 编译器功能综合测试用例
// 包含以下功能测试：
// 功能1: do-while循环
// 功能2: switch语句
// 功能3: 带参数和返回值的函数
// 功能4: 变量的作用域管理
// 功能5: 全局变量（在所有函数外部声明）
// 功能6: 多变量声明和初始化
// 功能7: 一维数组
// 功能8: 逻辑运算符 (&&, ||)
// 功能9: 关系运算符 (<, <=, >, >=, ==, !=)
// ============================================================

// 功能5: 全局变量（在所有函数外部声明）
var global_var:int = 100;

// 功能3: 带参数和返回值的函数
func add(a:int, b:int):int {
    var temp:int = a + b;
    return temp;
}

// 功能3: 带参数和返回值的函数
func multiply(a:int, b:int):int {
    var temp:int = a * b;
    return temp;
}

// 功能2: switch语句测试
func test_switch(n:int):int {
    var result:int = 0;
    switch(n) {
        case 1: result = 100;
        case 2: result = 200;
        case 5: result = 500;
        default: result = 999;
    }
    return result;
}

// 功能7: 一维数组测试
func fill_array(index:int):int {
    var arr[4]:int;
    arr[0] = 0;
    arr[1] = 5;
    arr[2] = 10;
    arr[3] = 15;
    return arr[index];
}

// 主函数：执行所有功能测试
main() {
    // 功能6: 变量声明和初始化
    var num1:int = 10;
    var num2:int = 20;
    var num3:int = 30;
    
    // 测试开始标记
    write 1000;
    
    // 功能5: 全局变量测试
    write global_var;
    global_var = global_var + 50;
    write global_var;
    
    // 功能6: 变量初始化测试
    write num1;
    write num2;
    write num3;
    
    // 功能3: 带参数和返回值的函数调用
    var sum_result:int = add(num1, num2);
    write sum_result;
    
    var product_result:int = multiply(num2, num3);
    write product_result;
    
    // 功能1: do-while循环测试
    var counter:int = 0;
    do {
        write counter;
        counter = counter + 1;
    } while (counter < 4);
    
    // 测试算术运算
    var total:int = sum_result + product_result;
    write total;
    
    // 功能7: 一维数组测试
    write fill_array(0);
    write fill_array(2);
    
    // 功能2: switch语句测试
    write test_switch(1);   // case 1 → 输出100
    write test_switch(2);   // case 2 → 输出200
    write test_switch(5);   // case 5 → 输出500
    write test_switch(100);  // default → 输出999
    
    // 功能8: 逻辑运算符测试 (&&, ||)
    var x:int = 5;
    var logic_test1:int = 0;
    var logic_test2:int = 0;
    
    // 测试 && (逻辑与)
    if (x < 10 && x > 0) {
        logic_test1 = 1;
    } else {
        logic_test1 = 0;
    }
    write logic_test1;
    
    // 测试 || (逻辑或)
    if (x < 0 || x > 0) {
        logic_test2 = 1;
    } else {
        logic_test2 = 0;
    }
    write logic_test2;
    
    // 功能9: 关系运算符测试 (<, <=, >, >=, ==, !=)
    var a:int = 10;
    var b:int = 20;
    
    // 测试 < (小于)
    if (a < b) {
        write 1;
    } else {
        write 0;
    }
    
    // 测试 <= (小于等于)
    if (a <= b) {
        write 1;
    } else {
        write 0;
    }
    
    // 测试 > (大于)
    if (b > a) {
        write 1;
    } else {
        write 0;
    }
    
    // 测试 >= (大于等于)
    if (b >= a) {
        write 1;
    } else {
        write 0;
    }
    
    // 测试 == (等于)
    if (a == a) {
        write 1;
    } else {
        write 0;
    }
    
    // 测试 != (不等于)
    if (a != b) {
        write 1;
    } else {
        write 0;
    }
    
    // 测试结束标记
    write 9999;
}
